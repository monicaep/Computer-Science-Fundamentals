	Making coffee:
	FUNCTION makeCoffee(cups)
	   POUR water into the coffee maker reservoir up the the line that is EQUAL TO the number of cups of coffee
	   INSERT paper filter into compartment
	   SCOOP an equal amount of scoops as desired cups of coffee
	   PRESS the on button
	END FUNCTION

	Washing dishes:
	FUNCTION wash dishes(dish)
	   PUT soap on sponge
		 WHILE dish is dirty
		 	SCRUB dish with soapy sponge
		 END WHILE
		 RINSE dish
		 PUT dish on drying rack
	END FUNCTION

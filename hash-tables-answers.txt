1. What is a hash table?
A hash table is a data structure that stores data in key-value pairs. The value is the data and the key is a unique number given to each value.

2. What is hashing?
Keys are usually strings. Hashing converts that string (or whatever else the key may be) to a fixed-length number (hash code) so that it can be easily accessed.

3. How does a hash table store data?
The data(value) is paired with a unique identifier (key). The value can then be accessed through the unique key. For example, data can be stored using a client's unique ID at a gym. The client's ID can serve as the key and the value can include account information like the client name, when their membership expires, and past visits to the gym. Gym employees can then use the client ID to look up their account information.

4. How are hash tables and objects different?
Hash tables allow users to access data using a unique identifier. Each bucket in a hash table will contain the same information (consistently formatted values). An object does not use a unique identifier to store values and although the values can also be consistently formatted, it is not required. A hash table also requires a hashing function to convert the unique identifiers to indices that can be easily accessed in an array.

6. Build a system that allows a sales associate to enter a customer's name, address, and phone number into the system and look up customers using their phone numbers. Store this information in a hash table.

```
function hash(string, capacity) {
  let hash = 0;
  for (let i = 0; i < string.length; i++) {
    hash += string.charCodeAt(i);
  }
  return hash % capacity;
};

class Customer {
  constructor(phone, name, address) {
    this.phone = phone;
    this.name = name;
    this.address = address;
  }
}

class HashTable {
  constructor() {
    this.bucketCount = 1000;
    this.buckets = [];
  }

  getBucketIndex(phone) {
    return hash(phone, this.bucketCount);
  }

  getBucket(phone) {
    this.buckets[this.getBucketIndex(phone)];
  }

  addCustomer(phone, name, address) {
    let newCustomer = new Customer(phone, name, address);
    this.buckets[this.getBucketIndex(phone)] = newCustomer;
  }

  getCustomer(phone) {
    return ('Name: ' + this.buckets[this.getBucketIndex(phone)].name + ' Address: ' + this.buckets[this.getBucketIndex(phone)].address);
  }
}

let customers = new HashTable();
customers.addCustomer('(123)123-1233', 'Molly', '123 S');
customers.addCustomer('(222)222-2222', 'Bob', '456 E');
customers.addCustomer('(567)888-343', 'Sally', '567 N');

console.log(customers.getCustomer('(123)123-1233'));
console.log(customers.getCustomer('(222)222-2222'));
console.log(customers.getCustomer('(567)888-343'));
```
7. Build a system that allows a store owner to track their store's inventory using a hash table for storage.

```
function hash (string, capacity) {
  let hash = 0;
  for (let i = 0; i < string.length; i++) {
    hash += string.charCodeAt(i);
  }
  return hash % capacity;
};

class Item {
  constructor(sku, name, amount) {
    this.sku = sku;
    this.name = name;
    this.amount = amount;
  }
};

class HashTable {
  constructor() {
    this.bucketCount = 1000;
    this.buckets = [];
  }

  getBucketIndex(sku) {
    return hash(sku, this.bucketCount);
  }

  getBucket(sku) {
    this.buckets[this.getBucketIndex(sku)];
  }

  addNewItem(sku, name, amount) {
    let newItem = new Item(sku, name, amount);
    this.buckets[this.getBucketIndex(sku)] = newItem;
  }

  getItem(sku) {
    return ('Name: ' + this.buckets[this.getBucketIndex(sku)].name + ' Amount: ' + this.buckets[this.getBucketIndex(sku)].amount);
  }

  restockItem(sku, addAmount) {
    this.buckets[this.getBucketIndex(sku)].amount += addAmount;
  }

  sellItem(sku, sellAmount) {
    this.buckets[this.getBucketIndex(sku)].amount -= sellAmount;
  }
}

let inventory = new HashTable();
inventory.addNewItem('A1', 't-shirt', 25);
inventory.addNewItem('B2', 'skinny jeans', 20);
inventory.addNewItem('C3', 'sneakers', 15);

//console.log(inventory);

inventory.restockItem('A1', 3);
console.log(inventory.getItem('A1'));

console.log(inventory.getItem('B2'));

console.log(inventory.getItem('B2'));
```

8. Build a system that allows digital copies of newspapers to be entered and searched by publisher and publication date. Use hash tables to store the necessary data.

```
function hash (publisher, date, capacity) {
  let hash = 0;
  let string = publisher.concat('', date);
  for (let i = 0; i < string.length; i++) {
    hash += string.charCodeAt(i);
  }
  return hash % capacity;
}

class Newspaper {
  constructor(publisher, date, paper) {
    this.publisher = publisher;
    this.date = date;
    this.paper = paper;
  }
}

class HashTable {
  constructor() {
    this.bucketCount = 10;
    this.buckets = [];
  }

  getBucketIndex(publisher, date) {
    return hash(publisher, date, this.bucketCount);
  }

  addPaper(publisher, date, paper) {
    let newPaper = new Newspaper(publisher, date, paper);
    this.buckets[this.getBucketIndex(publisher, date)] = newPaper;
  }

  findPaper(publisher, date) {
    return this.buckets[this.getBucketIndex(publisher, date)];
  }
}

let newspapers = new HashTable();
newspapers.addPaper('LV Review Journal', '09-25-1993', 'Heavy Traffic Due to Major Road Work');
newspapers.addPaper('News Media', '10-17-1990', 'Tornado in Kansas');
newspapers.addPaper('Daily Prophet News', '01-29-1997', 'Weasley wins the lottery');

//console.log(newspapers);

console.log(newspapers.findPaper('LV Review Journal', '09-25-1993'));
console.log(newspapers.findPaper('News Media', '10-17-1990'));
console.log(newspapers.findPaper('Daily Prophet News', '01-29-1997'));
```
